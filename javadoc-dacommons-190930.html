                                  public static void bootstrap(RequestDispatcher<DomainAgentRequestEntity, DomainAgentResponseEntity> requestDispatcher, Collection<IntentHandlerConfigurer> intentHandlerConfigurers)

                                  {
    if (requestDispatcher.isBootstraped()) {
        throw new AlreadyBootstrapedException();
    } 
    final IntentHandlerContext intentHandlerContext = new IntentHandlerContextImpl();
    intentHandlerConfigurers.forEach(intentHandlerConfigurer -> tryAction(() -> intentHandlerConfigurer.configureIntentHandler(intentHandlerContext)));
    if (intentHandlerContext.getRequestDispatcherInitializers().isEmpty()) {
        intentHandlerContext.addRequestDispatcherInitializer(new StdRequestDispatcherInitializer());
    } 
    intentHandlerContext.getRequestDispatcherInitializers().forEach(requestDispatcherInitializer -> tryAction(() -> requestDispatcherInitializer.init(requestDispatcher, intentHandlerContext)));
    intentHandlerContext.getFeatureConfigurers().stream().peek(featureConfigurer -> tryAction(() -> featureConfigurer.configureFeature(intentHandlerContext))).forEach(requestDispatcher::addHandlerMapperConfigurer);
    intentHandlerConfigurers.forEach(requestDispatcher::addHandlerMapperConfigurer);
    requestDispatcher.bootstrap();
}

                                  @Override public Object resolveArgument(MethodParameter parameter, DomainAgentRequestEntity daHttpRequest)
                                  {
    if (log.isDebugEnabled()) {
        log.debug("start");
    } 
    Objects.requireNonNull(parameter);
    Objects.requireNonNull(daHttpRequest);
    final DomainAgentRequest daRequest = daHttpRequest.getRequest();
    if (daRequest == null) {
        if (log.isTraceEnabled()) {
            log.trace("da request data 가 입력되지 않았습니다.");
        } 
        return null;
    } 
    final UserRequest userRequest = daRequest.getUserRequest();
    if (userRequest == null) {
        if (log.isTraceEnabled()) {
            log.trace("da request data에 userRequest 가 입력되지 않았습니다.");
        } 
        return null;
    } 
    final UserRequestParams params = userRequest.getParams();
    if (params == null) {
        if (log.isTraceEnabled()) {
            log.trace("da request data에 userRequest.params 가 입력되지 않았습니다.");
        } 
        return null;
    } 
    final List<State<?>> states = params.getState();
    if (states == null) {
        if (log.isTraceEnabled()) {
            log.trace("da request data에 userRequest.params.state 가 입력되지 않았습니다.");
        } 
        return null;
    } 
    if (states.isEmpty()) {
        if (log.isTraceEnabled()) {
            log.trace("da request data에 userRequest.params.state 가 비어 있습니다.");
        } 
        return null;
    } 
    Optional<String> tokenOption = states.stream().filter(state -> AudioPlayerContentState.TYPE.equals(state.getType())).map(state -> ((AudioPlayerContentState)state).getBody()).map(state -> state.getToken()).findFirst();
    if (!tokenOption.isPresent()) {
        tokenOption = states.stream().filter(state -> AudioPlayerPlayState.TYPE.equals(state.getType())).map(state -> ((AudioPlayerPlayState)state).getBody()).map(state -> state.getToken()).findFirst();
    } 
    return tokenOption.map(rawToken -> AudioTokenBuilder.build(rawToken)).orElseGet(() -> {
        if (log.isTraceEnabled()) {
            log.trace("da request data에 userRequest.params.state에 AudioPlayer.PlayState 혹은 AudioPlayer.ContentState 가 정의되어 있지 않습니다.");
        } 
        return null;
    });
}

                                  {

    final Class<?> parameterType = parameter.getParameterType();

    try {
        noArgConstr = parameterType.getConstructor();
    } catch (NoSuchMethodException e) {
        throw new RuntimeException();
    }
    if (noArgConstr == null) {
        throw new RuntimeException("기본 생성자가 필요합니다. parameterType: " + parameterType);
    }

    try {
        parameterObj = parameterType.getConstructor().newInstance();
    } catch (InstantiationException e) {
        e.printStackTrace();
        throw new RuntimeException();
    } catch (IllegalAccessException e) {
        e.printStackTrace();
        throw new RuntimeException();
    } catch (InvocationTargetException e) {
        e.printStackTrace();
        throw new RuntimeException();
    } catch (NoSuchMethodException e) {
        e.printStackTrace();
        throw new RuntimeException();
    }
    final DomainAgentRequest daRequest = daHttpRequest.getRequest();
    if (daRequest == null) {
        return parameterObj;
    } 
    final Action action = daRequest.getAction();
    if (action == null) {
        return parameterObj;
    } 
    final Map<String, String> params = action.getParams();
    if (params == null) {
        return parameterObj;

        return parameterObj;

        } finally {
            field.setAccessible(accessible);
        }
    }

                                  @Override public boolean isMatchingCondition(DomainAgentRequestEntity daHttpRequest)

        return intentParamsHint.strict();
    } 
    final DomainAgentRequest daRequest = daHttpRequest.getRequest();
    if (daRequest == null) {
        return false;

    final Action action = daRequest.getAction();

        return false;
    } 
    final Map<String, String> params = action.getParams();
    if (params == null) {
        return false;
    } 
    if (intentParamsHint.strict()) {
        if (requiredEntityNames.length != params.size()) {
            return false;
        } 
    } 
    boolean isMatch = true;
    for (int i = 0; i < requiredEntityNames.length; i++) {
        if (!params.containsKey(requiredEntityNames[i])) {
            isMatch = false;
            break;
        } 
    }
    return isMatch;
}
